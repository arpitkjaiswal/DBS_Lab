13.select distinct s1.course_id 
from section s1
where semester = 'Fall' and year = 2009
and exists (
    select 1 
    from section s2
    where s2.course_id = s1.course_id 
    and semester = 'Spring' and year = 2010
);

from student s
where not exists (
    select course_id 
    from course 
    where dept_name = 'Biology'
    minus
    select course_id 
    from takes 
    where ID = s.ID
);

17.select dept_name, avg_salary
from (
    select dept_name, avg(salary) as avg_salary
    from instructor
    group by dept_name
) dept_avg
where avg_salary > 42000;


18.create view all_courses as
select course_id, sec_id, building, room_number
from section
where course_id in (
    select course_id from course where dept_name = 'Physics'
)
and semester = 'Fall' and year = 2009;


19.select * from all_courses;

20.create view department_total_salary as
